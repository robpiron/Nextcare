        const sampleId = parseInt(document.getElementById('res-sample-id').value);

        const samTests = tests.filter(t => t.sample_id === sampleId);
        
        samTests.forEach(t => {
            const inputVal = document.getElementById(`test-input-${t.id}`).value;
            t.result_value = inputVal;
            
            const range = t.reference_range ? t.reference_range.split('-') : [];
            if (range.length === 2) {
                const min = parseFloat(range[0]);
                const max = parseFloat(range[1]);
                const val = parseFloat(inputVal);
                if (val < min || val > max) t.status = 'flagged';
                else t.status = 'completed';
            } else {
                t.status = 'completed';
            }
            t.verified_by = 5;
            t.verified_at = new Date().toISOString();
        });
        
        DB.set('lab_tests', tests);

        const updatedSam = DB.update('lab_samples', sampleId, { status: 'completed', report_notes: reportNotesHTML });

        // Trigger Email Notification
